#Thranduil recieves request from Elrond
narrative_event = {
    id = gfe.01
    title = "EVTNAMEgfe.01"
    desc = "EVTDESCgfe.01"
    picture = "GFX_shadows_coming"

	is_triggered_only = yes
	
    option = { 
		name = "EVTOPTAgfe.01" #Thranduil decides who to send
		
		trigger = {
			any_child = {
				if = {
					age = 30
				}
			}
		}
		
		any_child = {
			if = {
				age = 30
			}
			character_event = { id = gfe.08 }
			set_global_flag = wotrthrsent
		}
		
		ai_chance = {
			factor = 80
		}
	}
	
	option = { 
		name = "EVTOPTAgfe.01" #Thranduil decides who to send
		
		any_realm_character = {
			if = {
				OR = {
					has_character_flag = tauriel
					martial = 10
				}
			}
			character_event = { id = gfe.09 }
			set_global_flag = wotrthrsent
		}
		ai_chance = {
			factor = 15
		}
	}

    option = {
		name = "EVTOPTCgfe.01" #Thranduil sends no one
		
		trigger = {
			has_game_rule = {
				name = RULE_WOTR
				value = sandbox
			}
		}
		
		ai_chance = {
			factor = 5
		}
    }
}
