	is_triggered_only = yes

	    any_playable_ruler = {

			    culture_group = culture_group_quendi
				culture_group = culture_group_amanyar
				culture_group = culture_group_umanyar
			}
			character_event = { id = elven_dwarven.7 }
		}
	}
	
	option = {
	    name = "EVTOPTAelven_dwarven.5" # Good
		if = {
			limit = { NOT = { yearly_income = 40 } }
			wealth = 100
		}
		if = {
			limit = { 
				NOT = { yearly_income = 200 } 
				yearly_income = 40
			}
			scaled_wealth = 2.5
		}
		if = {
			limit = { yearly_income = 200 }
			wealth = 500
		}
		prestige = 150
		build_time_modifier = 
		clr_character_flag = trading_with_dwarves
		clr_character_flag = trading_with_elves
	}
}
